569 template<typename Class>
570 typename Vector<typename container_data<Class>::type>::resize_function_type DefaultVectorResizer(Class& input)
571 {
572  static_assert (is_resizable<Class>::value, "Class type is not resizable.");
573  static_assert (is_span_constructible<Class>::value || (is_data_available<Class>::value && is_size_available<Class>::value),
574  "Cannot create a span given the provided class.");
575 
576  using value_type = typename container_data<Class>::type;
577 
578  if constexpr (is_span_constructible<Class>::value)
579  {
580  using index_type = typename Vector<value_type>::index_type;
581  using resize_function = typename Vector<value_type>::resize_function_type;
582 
583  Class* inputPtr = &input;
584  resize_function resizeLambda =
585  [inputPtr](index_type newSize) -> iDynTree::Span<typename Class::value_type>
586  {
587  inputPtr->resize(newSize);
588  return iDynTree::make_span(*inputPtr);
589  };
590 
591  return resizeLambda;
592  }
593  else
594  {
595  using index_type = decltype(std::declval<Class>().size());
596  using resize_function = typename Vector<value_type>::resize_function_type;
597 
598  Class* inputPtr = &input;
599  resize_function resizeLambda =
600  [inputPtr](index_type newSize) -> iDynTree::Span<value_type>
601  {
602  inputPtr->resize(newSize);
603  return iDynTree::make_span(inputPtr->data(), inputPtr->size());
604  };
605 
606  return resizeLambda;
607  }
608 }

619 template<typename Class>
620 Vector<typename container_data<Class>::type>
621 make_vector(Class& input, VectorResizeMode mode = VectorResizeMode::Fixed)
622 {
623  static_assert (!std::is_same<typename container_data<Class>::type, bool>::value,
624  "Cannot create a Vector of bool type. Memory is not contiguos." );
625  static_assert (is_span_constructible<Class>::value || (is_data_available<Class>::value && is_size_available<Class>::value),
626  "Cannot create a span given the provided class.");
627 
628  using value_type = typename container_data<Class>::type;
629 
630  iDynTree::Span<value_type> span;
631 
632  if constexpr (is_span_constructible<Class>::value)
633  {
634  span = iDynTree::make_span(input);
635  }
636  else
637  {
638  span = iDynTree::make_span(input.data(), input.size());
639  }
640 
641  if constexpr (is_resizable<Class>::value)
642  {
643  if (mode == VectorResizeMode::Resizable)
644  {
645  return Vector(span, DefaultVectorResizer(input));
646  }
647  else
648  {
649  return Vector(span);
650  }
651  }
652  else
653  {
654  if (mode == VectorResizeMode::Resizable)
655  {
656  std::cerr << "[GenericContainer::Vector] " << type_name<Class>()
657  << " is not resizable. Returning a non-resizable container." << std::endl;
658  }
659 
660  return Vector(span);
661  }
